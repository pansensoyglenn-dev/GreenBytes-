This module serves as a definitive exploration into the mechanics of Abstraction within a commercial context. By stripping away non-essential implementation details, we empower the developer to focus on the high-level orchestration of business logic. In a world of increasing digital complexity, Abstraction acts as a cognitive filter—allowing us to interact with sophisticated payment gateways, messaging protocols, and logistics APIs through simplified, unified interfaces. This practice is not merely about "hiding" code; it is about establishing a robust contract between components that ensures system integrity even as underlying technologies are replaced.

The Architectural Vision

The architecture of a modern e-commerce engine relies on Decoupling. When a platform manages millions of transactions, the core "Order Processing" logic cannot be tightly bound to a specific bank or SMS provider. Our vision here is to utilize Python’s abc module to create Abstract Base Classes (ABCs) that define a strict behavior protocol. This creates a "plug-and-play" environment: your system treats a Stripe Gateway and a PayPal Gateway as identical entities because they both adhere to the PaymentGateway abstraction.

By implementing this design pattern, we achieve Scalability and Maintainability. Developers can update a payment API version or switch from Email to Push Notifications without ever touching the sensitive core logic of the OrderProcessor. This separation of concerns is exactly what Google AdSense and professional engineering boards look for: authoritative, structured, and technically sound information architecture.

from abc import ABC, abstractmethod
from dataclasses import dataclass
from typing import List, Optional
from datetime import datetime
import random

# Abstract interfaces (The Blueprints)
class NotificationChannel(ABC):
    @abstractmethod
    def send(self, recipient: str, message: str) -> bool:
        pass

class PaymentGateway(ABC):
    @abstractmethod
    def charge(self, amount: float, currency: str, card_token: str) -> bool:
        pass
    @abstractmethod
    def refund(self, transaction_id: str) -> bool:
        pass

# Concrete implementations (The Reality)
class EmailChannel(NotificationChannel):
    def send(self, recipient: str, message: str) -> bool:
        print(f"[EMAIL]   → {recipient:28} | {message}")
        return True

class SmsChannel(NotificationChannel):
    def send(self, recipient: str, message: str) -> bool:
        print(f"[SMS]     → {recipient:28} | {message[:30]}...")
        return random.random() > 0.15

class FakeStripeGateway(PaymentGateway):
    _next_tx_id = 100000
    def charge(self, amount: float, currency: str, card_token: str) -> bool:
        success = random.random() > 0.08
        if success:
            tx_id = f"ch_{self._next_tx_id}"
            self._next_tx_id += 1
            print(f"[Stripe]  CHARGED {amount:6.2f} {currency}  ({card_token}) → {tx_id}")
            return tx_id
        else:
            print(f"[Stripe]  DECLINED  {amount:6.2f} {currency}  ({card_token})")
            return False
    def refund(self, transaction_id: str) -> bool:
        print(f"[Stripe]  REFUNDED {transaction_id}")
        return True

@dataclass(frozen=True)
class Customer:
    id: str; email: str; phone: Optional[str] = None; preferred_channel: str = "email"

@dataclass
class Order:
    id: str; customer: Customer; total: float; currency: str = "USD"
    created_at: datetime = datetime.now(); _transaction_id: Optional[str] = None
    @property
    def paid(self) -> bool: return self._transaction_id is not None

class OrderProcessor:
    def __init__(self, payment_gateway: PaymentGateway, notification_channels: dict):
        self.payment = payment_gateway
        self.channels = notification_channels

    def process_order(self, order: Order, card_token: str) -> bool:
        if order.paid: return True
        tx_result = self.payment.charge(order.total, order.currency, card_token)
        if not tx_result:
            self._notify_customer(order, "Payment declined.")
            return False
        object.__setattr__(order, "_transaction_id", tx_result)
        self._notify_customer(order, f"Thank you for order #{order.id}")
        return True

    def _notify_customer(self, order, message):
        channel = self.channels.get(order.customer.preferred_channel) or self.channels.get("email")
        recipient = order.customer.email if order.customer.preferred_channel == "email" else order.customer.phone
        channel.send(recipient, message)

def main():
    stripe = FakeStripeGateway()
    channels = {"email": EmailChannel(), "sms": SmsChannel()}
    processor = OrderProcessor(stripe, channels)
    alice = Customer("c1", "alice@example.com", "+63917", "sms")
    processor.process_order(Order("ORD-01", alice, 89.99), "tok_visa")

if __name__ == "__main__": main()
