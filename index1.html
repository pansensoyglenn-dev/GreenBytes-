class Employee:
    def __init__(self, employee_id: str, name: str, base_salary: float, department: str):
        self.__employee_id = employee_id      # Truly private – nobody should change this
        self.__name = name                    # Truly private – legal name shouldn't be modified casually
        self._base_salary = base_salary       # Protected – subclasses may need to read/adjust it
        self.__department = department
        self.__performance_bonus = 0.0
        self.__tax_rate = 0.15

    def _calculate_deductions(self) -> float:
        """Protected method: Calculates deductions like taxes. Subclasses can override."""
        return self._base_salary * self.__tax_rate

    def add_bonus(self, bonus_amount: float) -> None:
        if bonus_amount < 0:
            raise ValueError("Bonus amount cannot be negative.")
        self.__performance_bonus += bonus_amount
        print(f"Bonus of ${bonus_amount:.2f} added to {self.__name}.")

    def get_net_salary(self) -> float:
        gross = self._base_salary + self.__performance_bonus
        deductions = self._calculate_deductions()
        return gross - deductions

    def get_employee_info(self) -> str:
        return f"ID: {self.__employee_id}, Name: {self.__name}, Dept: {self.__department}"

class Manager(Employee):
    def __init__(self, employee_id: str, name: str, base_salary: float, department: str, team_size: int):
        super().__init__(employee_id, name, base_salary, department)
        self.__team_size = team_size
        self._management_bonus_rate = 0.05

    def _calculate_deductions(self) -> float:
        return self._base_salary * 0.20

    def calculate_management_bonus(self) -> float:
        bonus = self._base_salary * self._management_bonus_rate * (self.__team_size / 10)
        self.add_bonus(bonus)
        return bonus

class PayrollSystem:
    def __init__(self):
        self.__employees = {}

    def add_employee(self, employee: Employee) -> None:
        emp_id = employee._Employee__employee_id 
        if emp_id in self.__employees:
            raise ValueError("Employee ID already exists.")
        self.__employees[emp_id] = employee
        print(f"Added: {employee.get_employee_info()}")

    def process_payroll(self) -> dict:
        total = 0.0
        for emp in self.__employees.values():
            if isinstance(emp, Manager):
                emp.calculate_management_bonus()
            total += emp.get_net_salary()
        return {"Total": total}

if __name__ == "__main__":
    payroll = PayrollSystem()
    payroll.add_employee(Employee("E01", "Alice", 60000, "Dev"))
    payroll.add_employee(Manager("M01", "Bob", 80000, "Ops", 10))
    print(f"\nPayroll Processed: ${payroll.process_payroll()['Total']:.2f}")

Step-by-Step Logical Breakdown

1. Employee Class – The Core Protected Entity

Logical reasoning: Double underscore (__) = name mangling. Strongest privacy signal in Python. Real payroll systems must protect identity and financial data.

Why protected (_) instead of private (__)? Allows responsible extension by subclasses while still signaling: "Don't touch unless you really know what you're doing."

Encapsulation win: All bonus logic + validation lives in one place.

2. Manager Class – Responsible Extension

Different tax treatment is very realistic. Protected methods allow clean overrides without breaking the contract.

3. PayrollSystem – High-level Business Logic

Private employee registry + usage of public API only = very robust design.
