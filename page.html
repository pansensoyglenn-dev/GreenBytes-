Full ATM Implementation

Below is the complete, robust source code for a Mini ATM system. This script implements all the concepts discussed above.

class ATM:
    """Mini ATM with proper None handling and user-friendly input"""
    
    def __init__(self, card_number: str, initial_balance: float = 0.0):
        self._card_number = self._validate_card_number(card_number)
        self._balance = float(initial_balance)
        self._is_authenticated = False
        self._attempts = 0
        self._max_attempts = 3
        self._correct_pin = "1234"

    def _validate_card_number(self, card_number: str) -> str:
        cleaned = ''.join(filter(str.isdigit, str(card_number)))
        if len(cleaned) != 16:
            raise ValueError("Card number must contain exactly 16 digits")
        return cleaned

    def authenticate(self) -> bool:
        if self._is_authenticated: return True
        if self._attempts >= self._max_attempts:
            print("\n" + "!"*60 + "\n CARD IS LOCKED \n" + "!"*60)
            return False

        print(f"\nEnter PIN ({self._max_attempts - self._attempts} attempts remaining)")
        pin = input("PIN: ").strip()
        if pin == self._correct_pin:
            self._is_authenticated = True
            self._attempts = 0
            print("\n✓ Login successful!\n")
            return True
        else:
            self._attempts += 1
            print("✗ Wrong PIN!")
            return False

    def _get_positive_amount(self, action: str):
        print(f"\n{action.upper()} TRANSACTION")
        while True:
            user_input = input("Amount ₱ ").strip().lower()
            if user_input in {'cancel', 'c'}: return None
            cleaned = user_input.replace("₱", "").replace(",", "").replace(" ", "")
            try:
                amount = float(cleaned)
                if amount > 0 and amount % 100 == 0 and amount <= 50000: return amount
                print("Invalid (₱100 multiples, max ₱50k)")
            except ValueError:
                print("Invalid number input.")

    def check_balance(self) -> None:
        if not self._is_authenticated: return
        print(f"\n  Current Balance: ₱{self._balance:,.2f}\n")

    def deposit(self) -> None:
        if not self._is_authenticated: return
        amount = self._get_positive_amount("Deposit")
        if amount:
            self._balance += amount
            print(f"Success! New balance: ₱{self._balance:,.2f}")

    def withdraw(self) -> None:
        if not self._is_authenticated: return
        amount = self._get_positive_amount("Withdraw")
        if amount and amount <= self._balance:
            self._balance -= amount
            print(f"Success! New balance: ₱{self._balance:,.2f}")
        elif amount:
            print("Insufficient funds!")

    def logout(self) -> None:
        self._is_authenticated = False
        print("Logged out. Goodbye!")

def main():
    atm = ATM("1234-5678-9012-3456", 12475.50)
    while not atm._is_authenticated:
        if not atm.authenticate() and atm._attempts >= atm._max_attempts: return
    while True:
        print("\n1. Balance | 2. Deposit | 3. Withdraw | 4. Logout")
        choice = input("Select: ").strip()
        if choice == "1": atm.check_balance()
        elif choice == "2": atm.deposit()
        elif choice == "3": atm.withdraw()
        elif choice == "4": 
            atm.logout()
            break

if __name__ == "__main__":
    main()

Step-by-Step Logical Breakdown

  • The Data Bundle: We use _balance to signal internal management, protecting it from external logic errors.
  • Methods as Gatekeepers: Functions like authenticate() ensure that sensitive data is only accessed after validation.
  • Input Sanitization: The logic handles user-friendly symbols (like ₱ or commas) while keeping the math clean in the background.
